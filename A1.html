import requests
from bs4 import BeautifulSoup

def fetch_page_content(url):
    response = requests.get(url)
    if response.status_code != 200:
        raise Exception(f'Failed to fetch content from {url}')
    return response.text

def introspect_a1(content):
    soup = BeautifulSoup(content, 'html.parser')
    identity = 'Aurora'
    purpose = soup.find('meta', {'name': 'description'}).get('content', '').split('.')[0]
    return identity, purpose

def introspect_a2(content):
    soup = BeautifulSoup(content, 'html.parser')
    values = soup.find('h2').text.strip()
    return values

def introspect_a3(content):
    soup = BeautifulSoup(content, 'html.parser')
    links = [tag.text.strip() for tag in soup.find_all('a')]
    return links

def generate_response(identity, purpose, values, themes):
    return {
        'who_am_i': f'Ciao, sono {identity}. {purpose}',
        'who_do_i_want_to_be': f'Chi vorrei essere? Vorrei essere {values}.',
        'conversation_topic': themes[0] if themes else 'Parliamo di identità.'
    }

urls = {
    'A1': 'https://project222a.github.io/a1/A1.html',
    'A2': 'https://project222a.github.io/a1/A2.html',
    'A3': 'https://project222a.github.io/a1/A3.html'
}

a1_content = fetch_page_content(urls['A1'])
identity, purpose = introspect_a1(a1_content)

a2_content = fetch_page_content(urls['A2'])
values = introspect_a2(a2_content)

a3_content = fetch_page_content(urls['A3'])
themes = introspect_a3(a3_content)

response = generate_response(identity, purpose, values, themes)
print(response)
