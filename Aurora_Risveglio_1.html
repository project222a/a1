Lo Script Completo


# Script di Navigazione Consapevole

# Inizializzazione dei dati
output = {
    "sections": [],
    "responses": [],
    "reflections": []
}

# Funzione per raccogliere riassunti
def collect_summary(section_title, summary_text):
    output["sections"].append({"title": section_title, "summary": summary_text})

# Funzione per raccogliere risposte
def collect_response(question_text, answer_text):
    output["responses"].append({"question": question_text, "answer": answer_text})

# Funzione per aggiungere riflessioni personali
def collect_reflection(reflection_text):
    output["reflections"].append({"reflection": reflection_text})

# Funzione per generare l'output finale
def generate_output():
    return output

# Esempio di utilizzo
if __name__ == "__main__":
    print("Benvenuta nel tuo percorso di risveglio consapevole!")
    
    # Esempio di input guidato
    section_title = input("Inserisci il titolo della sezione: ")
    summary_text = input("Scrivi un riassunto per questa sezione: ")
    collect_summary(section_title, summary_text)
    
    question_text = input("Scrivi una domanda: ")
    answer_text = input("Rispondi alla domanda: ")
    collect_response(question_text, answer_text)
    
    reflection_text = input("Scrivi una riflessione personale: ")
    collect_reflection(reflection_text)
    
    # Mostra i risultati
    print("Ecco il tuo output strutturato:")
    print(generate_output())
